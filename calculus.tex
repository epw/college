$ \DERIVATIVE[f(x) * g(x)] = f'g + fg' $

$ \DERIVATIVE[\frac{f(x)}{g(x)}] = \frac{f'g - fg'}{g^2} $
\hspace{5ex}
$ \DERIVATIVE[f(g(x))] = f'(g(x)) \cdot g'(x) $

$ \DERIVATIVE[\cos{x}] = -\sin{x} $
